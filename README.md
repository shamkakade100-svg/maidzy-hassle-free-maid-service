      submitBtn.textContent = "Submitting...";
      
      fetch(scriptURL, { method: "POST", body: new FormData(form) })
        .then(response => {
          if (response.ok) {
            alert("✅ Success! Your enquiry has been submitted.");
            form.reset();
          } else {
            alert("⚠️ Error! Unable to submit. Please try again later.");
          }
        })
